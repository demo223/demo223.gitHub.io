# Maven项目转换为SpringBoot项目

# 1、添加parent父级依赖

# 添加parent父级依赖

<!-- 这个parent是springboot的父级依赖，
    它提供相关的starter的maven管理以及版本号管理，还有相关maven插件的公共配置 -->
<parent>
    <artifactId>spring-boot-starter-parent</artifactId>
    <groupId>org.springframework.boot</groupId>
    <version>2.4.5</version>
    <relativePath></relativePath>
</parent>

# (可选)配置数据源，数据连接池

Spring Boot默认的数据源是：

  • 2.0之前：org.apache.tomcat.jdbc.pool.DataSource
  • 2.0及之后: com.zaxxer.hikari.HikariDataSource

配置druid数据源

spring:
  datasource:
    druid:
      driver-class-name: com.mysql.cj.jdbc.Driver
      url: jdbc:mysql://localhost:3306/crud?useSSL=false&&characterEncoding=utf-8
      username: root
      password: root

区别：

  • 第1种写法中，通过使用 Druid 的配置前缀 druid: 来配置数据源。在这种写法中，可以使用 Druid 提供的数据源配置项，如 initial-size、max-active、min-idle 等来配置数据源。
  • 第2种写法中，通过设置 type 属性为 com.alibaba.druid.pool.DruidDataSource 来使用 Druid 数据源。在这种写法中，可以使用 Spring Boot 提供的数据源配置项，如 driver-class-name、url、username、password 等来配置数据源。

这两种写法都是可以的，只是使用的配置项不同而已。

配置druid数据连接池，默认是使用Hikari

datasource:
    druid:
      driver-class-name: com.mysql.cj.jdbc.Driver
      url: jdbc:mysql://localhost:3306/crud?useSSL=false&&characterEncoding=utf-8
      username: root
      password: root
      initial-size: 5
      max-active: 20
      min-idle: 5
      max-wait: 60000
      time-between-eviction-runs-millis: 60000
      min-evictable-idle-time-millis: 300000
      validation-query: SELECT 1
      test-while-idle: true
      test-on-borrow: false
      test-on-return: false
      pool-prepared-statements: true
      max-open-prepared-statements: 20
      max-pool-prepared-statement-per-connection-size: 20
      filters: stat,wall,log4j
      connection-properties: druid.stat.mergeSql=true;druid.stat.slowSqlMillis=5000

# 5、添加application配置文件

server:
  port: 8888 # 将应用程序的端口设置为8080

spring:
  application:
    name: crud-demo # 将应用程序的名称设置为crud-demo
  datasource:
    driver-class-name: com.mysql.cj.jdbc.Driver
    url: jdbc:mysql://localhost:3306/crud?useSSL=false&&characterEncoding=utf-8
    username: root
    password: root

mybatis-plus:
  mapper-locations: /mapper/**/*.xml  # 实现mapper接口配置见mapper和接口的绑定
  configuration:
    map-underscore-to-camel-case: true  # 开启数据库实体类之间的驼峰映射
    log-impl: org.apache.ibatis.logging.stdout.StdOutImpl # 开启SQL日志打印
  global-config:
    db-config:
      table-prefix: tb_  #  设置数据库中表的前缀
      id-type: assign_id  # 默认使用雪花算法

logging:
  level:
    com.zq: debug # 将com.zq包下的所有日志输出级别设置为debug
  pattern:
    dateformat: MM-dd HH:mm:ss:SSS  # 日志输出的时间格式月-日 时:分:秒:毫秒
