# 实体类

Mybatis-Plus默认根据pojo类名找和类名同名的表，而我们的表名叫tb_user，所以不存在。

  • 解决1：在实体类上使用@TableName注解表名。
  • 解决2：在实际开发中，我们经常在让表名称以tb_开头，那么在每一个pojo类上使用@TableName注解指定表名表就非常麻烦 了，所以可以在springboot配置文件中配置表的前缀。
@Data
@AllArgsConstructor
@NoArgsConstructor
// @TableName("tb_user")    或者在配置文件中给所有的表添加table-prefix: tb_
public class User implements Serializable {
    private static final long serialVersionUID = 1L;

    private Long id;
    private String name;
    private String password;
    private int age;
    private Date birthday;
    @TableField(fill = FieldFill.INSERT)
    private LocalDateTime createTime;
    @TableField(fill = FieldFill.INSERT_UPDATE)
    private LocalDateTime updateTime;
}

# 主键存在的问题

JavaScript中Number类型并不能完全表示Long型的数字，在Long长度大于16位时会出现精度丢失的问题，超过17位的部分浏览器会转换为0显示，第17也可能会出现精度问题。

如果数据库中实体类需要转换的属性很多，我们就需要使用对象转换器

对象转换器JacksonObjectMapper是基于Jackson进行Java对象到json数据的转换

import static com.fasterxml.jackson.databind.DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES;
public class JacksonObjectMapper extends ObjectMapper {

    public static final String DEFAULT_DATE_FORMAT = "yyyy-MM-dd";
    public static final String DEFAULT_DATE_TIME_FORMAT = "yyyy-MM-dd HH:mm:ss";
    public static final String DEFAULT_TIME_FORMAT = "HH:mm:ss";

    public JacksonObjectMapper() {
        super();
        //收到未知属性时不报异常
        this.configure(FAIL_ON_UNKNOWN_PROPERTIES, false);
        //反序列化时，属性不存在的兼容处理
 							this.getDeserializationConfig().withoutFeatures(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
        SimpleModule simpleModule = new SimpleModule()
                .addDeserializer(LocalDateTime.class, new LocalDateTimeDeserializer(DateTimeFormatter.ofPattern(DEFAULT_DATE_TIME_FORMAT)))
                .addDeserializer(LocalDate.class, new LocalDateDeserializer(DateTimeFormatter.ofPattern(DEFAULT_DATE_FORMAT)))
                .addDeserializer(LocalTime.class, new LocalTimeDeserializer(DateTimeFormatter.ofPattern(DEFAULT_TIME_FORMAT)))
                .addSerializer(BigInteger.class, ToStringSerializer.instance)
                .addSerializer(Long.class, ToStringSerializer.instance)
                .addSerializer(LocalDateTime.class, new LocalDateTimeSerializer(DateTimeFormatter.ofPattern(DEFAULT_DATE_TIME_FORMAT)))
                .addSerializer(LocalDate.class, new LocalDateSerializer(DateTimeFormatter.ofPattern(DEFAULT_DATE_FORMAT)))
                .addSerializer(LocalTime.class, new LocalTimeSerializer(DateTimeFormatter.ofPattern(DEFAULT_TIME_FORMAT)));

        //注册功能模块 例如，可以添加自定义序列化器和反序列化器
        this.registerModule(simpleModule);
    }
}

然后在WebMvcConfig配置类中扩展MVC的消息转换器，重写extendMessageConverters方法

@Configuration
public class WebMvcConfig implements WebMvcConfigurer {
   /**
    * 扩展MVC框架的消息转换器
    * @param converters
    */
   @Override
   public void extendMessageConverters(List<HttpMessageConverter<?>> converters) {
      // 创建消息转换器对象
      MappingJackson2HttpMessageConverter messageConverter = new MappingJackson2HttpMessageConverter();
      // 设置对象转换器
      messageConverter.setObjectMapper(new JacksonObjectMapper());
      // 将上面的消息转换器对象追加到mvc框架的转换器集合中
      converters.add(0,messageConverter);
   }
}

# 批量删除

@DeleteMapping
public Result deleteOrderByIds(Long[] ids){
	LambdaQueryWrapper<Order> queryWrapper = new LambdaQueryWrapper<>();
	queryWrapper.in(Order::getId, Arrays.asList(ids));
	int count = orderService.count(queryWrapper);
	if(count != ids.length) {
		return Result.error("删除失败,有id不存在");
	}
	orderService.removeByIds(Arrays.asList(ids));
	return Result.success("删除成功");
}

# 动态修改

@PutMapping
public Result updateOrder(Order order) {
	LambdaQueryWrapper<Order> queryWrapper = new LambdaQueryWrapper<>();
	queryWrapper.eq(order.getId()!=null,Order::getId,order.getId());
	Order one = orderService.getOne(queryWrapper);
	if (one == null) {
	    return Result.error("订单不存在");
	}
	LambdaUpdateWrapper<Order> updateWrapper = new LambdaUpdateWrapper<>();
	updateWrapper.eq(order.getId()!=null,Order::getId, order.getId());
	orderService.update(order,updateWrapper);
	return Result.success("修改成功");
}

@GetMapping
public Result getOrder(String search) {
	if (StringUtils.isBlank(search)) {
		List<Order> list = orderService.list();
		return Result.success(list);
	}
	LambdaQueryWrapper<Order> queryWrapper = new LambdaQueryWrapper<>();
	queryWrapper.like(Order::getName,search);
	queryWrapper.or();
	queryWrapper.like(Order::getAddress,search);
//queryWrapper.and(qw>qw.like(Order::getName,search).or().like(Order::getAddress,search));
	List<Order> list = orderService.list(queryWrapper);
	return Result.success(list);

@GetMapping("page")
public Result page(String search){
   Page<Order> page = new Page<>(1,5);
   LambdaQueryWrapper<Order> queryWrapper = new LambdaQueryWrapper<>();
   queryWrapper.like(Order::getName,search);
   queryWrapper.orderByDesc(Order::getMoney);
   orderService.page(page, queryWrapper);
   HashMap<String, Object> map = new HashMap<>();
   // 分页数据
   List<Order> list = page.getRecords();
   // 分页条数
   long total = page.getTotal();
   map.put("data",list);
   map.put("total",total);
   map.put("pages",page.getPages());   //  总页数
   map.put("currentPage",page.getCurrent()); //  当前页码
   map.put("pageSize",page.getSize()); //  每页条数
   return Result.success(map);
}
